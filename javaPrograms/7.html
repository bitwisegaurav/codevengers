7. Write a Java program to convert a decimal number to binary numbers

Code :

class seven {
  public static void main(String[] args) throws java.io.IOException {
    java.io.BufferedReader input = new java.io.BufferedReader(new java.io.InputStreamReader(System.in));
    int decimalNumber;

    try {
      System.out.print("Enter a decimal number: ");
      decimalNumber = Integer.parseInt(input.readLine());
      
      java.lang.StringBuffer binary = new java.lang.StringBuffer();
      
      while (decimalNumber > 0) {
          binary.insert(0, decimalNumber % 2);
          decimalNumber /= 2;
      }
      
      System.out.println("Binary equivalent: " + binary.toString());
    } catch (java.lang.NumberFormatException e) {
      System.out.println("Invalid input, please enter a valid binary number.");
    } catch (Exception e) {
      System.out.println("Something is wrong ");
    }
  }
}
