94. Write a Java program to find second lowest number from the array

Code :

class ninetyFour {
    public static void main(String[] args) throws java.io.IOException {
        java.io.BufferedReader input = new java.io.BufferedReader(new java.io.InputStreamReader(System.in));

        try {
            System.out.print("Enter the elements of the array: ");
            String[] numbers = input.readLine().split(" ");

            java.util.ArrayList<Integer> arr = new java.util.ArrayList<>();
            
            for (String num : numbers) 
                arr.add(Integer.parseInt(num));
            
            if(arr.size()<2){
                System.out.println("At least two elements are required for finding the second largest.");
            } else {
                int lowest = Integer.MAX_VALUE, secondLowest = Integer.MAX_VALUE;

                for (int num : arr) {
                    if (num < lowest) {
                        secondLowest = lowest;
                        lowest = num;
                    } else if (num < secondLowest && num != lowest) {
                        secondLowest = num;
                    }
                }

                if (secondLowest == Integer.MAX_VALUE) {
                    System.out.println("There is no second lowest element.");
                } else {
                    System.out.println("Second lowest element: " + secondLowest);
                }
            }
        } catch (java.lang.NumberFormatException e) {
            System.out.println("Invalid input. Please enter valid numbers.");
        } catch (Exception e) {
            System.out.println("Something went wrong.");
        }
    }
}
