21. Write a Java program to swap the first and last elements of an array and create a new array

Code :

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;

class twentyone {
  public static void main(String[] args) throws java.io.IOException {
    BufferedReader input = new BufferedReader(new InputStreamReader(System.in));

    try {
      ArrayList<Integer> arr = new ArrayList<>();

      System.out.print("Enter the elements of the array : ");
      String a = input.readLine();
      String[] numbers = a.split(" ");

      for (String num : numbers) 
        arr.add(Integer.parseInt(num));

      if (arr.size() > 1) {
        ArrayList<Integer> swappedArr = new ArrayList<>();
        swappedArr.add(arr.get(arr.size() - 1));
        for (int i = 1; i < arr.size() - 1; i++) 
          swappedArr.add(arr.get(i));
        
        swappedArr.add(arr.get(0));

        System.out.print("Original Array : ");
        for (int num : arr) 
          System.out.print(num + " ");
        
        System.out.print("\nSwapped Array : ");
        for (int num : swappedArr) 
          System.out.print(num + " ");
        
        System.out.println();
      } else {
        System.out.println("Array should have at least 2 elements for swapping.");
      }
    } catch (NumberFormatException e) {
      System.out.println("Invalid input, please enter valid numbers.");
    } catch (Exception e) {
      System.out.println("Something went wrong.");
    }
  }
}
