50. Java Pogram To Find Distance Between Two Points

Code :

class fifty {
    public static void main(String[] args) throws java.io.IOException {
        java.io.BufferedReader input = new java.io.BufferedReader(new java.io.InputStreamReader(System.in));

        try {
            System.out.print("Enter the x-coordinate of point A: ");
            double x1 = Double.parseDouble(input.readLine());
            System.out.print("Enter the y-coordinate of point A: ");
            double y1 = Double.parseDouble(input.readLine());
            System.out.print("Enter the x-coordinate of point B: ");
            double x2 = Double.parseDouble(input.readLine());
            System.out.print("Enter the y-coordinate of point B: ");
            double y2 = Double.parseDouble(input.readLine());

            double distance = Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));
            System.out.println("Distance between points A and B: " + distance);
        } catch (NumberFormatException e) {
            System.out.println("Invalid input. Please enter valid numeric values.");
        } catch (Exception e) {
            System.out.println("Something went wrong.");
        }
    }
}
