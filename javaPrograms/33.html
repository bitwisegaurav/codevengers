33. Java Program to find area of Prism

Code :

class thirtythree {
    public static void main(String[] args) throws java.io.IOException {
        java.io.BufferedReader input = new java.io.BufferedReader(new java.io.InputStreamReader(System.in));

        try {
            System.out.print("Enter the base area of the prism: ");
            double baseArea = Double.parseDouble(input.readLine());
            System.out.print("Enter the height of the prism: ");
            double height = Double.parseDouble(input.readLine());

            if (baseArea < 0 || height < 0) {
                System.out.println("Please enter non-negative values for base area and height.");
            } else {
                double volume = baseArea * height;
                System.out.println("Volume of the prism: " + volume);
            }
        } catch (NumberFormatException e) {
            System.out.println("Invalid input. Please enter valid numeric values.");
        } catch (Exception e) {
            System.out.println("Something went wrong.");
        }
    }
}
