20. Write a Java program to compute the sum of the first 100 prime numbers

Code :

import java.util.ArrayList;

class twenty {
  public static void main(String[] args) {
    int count = 0, maxPrimeCount = 100;
    long sum = 0;
    ArrayList<Boolean> isPrime = new ArrayList<>();
    
    isPrime.add(false);
    isPrime.add(false);

    for (int i = 2; i < 1000; i++) isPrime.add(true);

    for (int i = 2; i*i < 1000; i++) 
      if (isPrime.get(i)) 
        for (int j = i; i*j < 1000; j++) 
          isPrime.set(i*j, false);

    System.out.println("First 100 primes: ");
    for (int i = 2; i < 1000; i++) {
      if (isPrime.get(i)) {
        sum += i;
        count++;
        System.out.print(i+", ");
        if (count == maxPrimeCount) break;
      }
    }

    System.out.println("\n");
    System.out.println("Sum of first 100 primes: " + sum);
  }
}
